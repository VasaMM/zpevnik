\hlavicka{Panenka}{}


\sloka{
\G Co skrýváš za \C víčky a \G plameny \C svíčky\\
snad \G houf bílych holubic, nebo jen \D žal,\\
tak \C odplul ten \G prvý, den \C zmáčený \G krví\\
ani \G pouťovou panenku \D nezane\G chal.
}

\refren{
\G Otevři \C oči Ty \G uspěcha\D ná,\\
\C dámo \G uplaka\D ná,\\
\C otevři \G oči ta \C hloupá noc \G končí\\
a \G mír je \D mezi ná\G ma.
}

\sloka{
\G Už si oblékni \C šaty a \G řetízek \C zlatý\\
a \G umyj se půjdeme na karne\D val\\
a na \C bílou \G kůži Ti \C napíšu \G tuší\\
že \G dámou jsi byla a \D zůstáváš \G dál.
}
