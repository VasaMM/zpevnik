\zvyraznit
\hlavicka[Arakain]{Zimní královna}

\sloka{
\GSiroky V pohád\D kách najdeš \Emi moudrou pravdu spousty \C příběhů\\
\GSiroky stejně i \D svůj v nich můžeš \C číst.\quad\X\\
\GSiroky Okouz\D lí cizí \Emi žena lásku tvou toho \C s kým jste si souzeni\\
\GSiroky tak jak to \D nést kde návod \C vzít.\quad\X
}

\refren{
\GSiroky Jenom si \D vzpomeň jednou \Emi přijela tak Zimní \C královna\\
\GSiroky malýho \D kluka odvezla \C na saních.\quad\X\\
\GSiroky Gerda \D pláče volá \Emi Káji, Káj má srdce \C studený\\
\GSiroky malá \D Gerda musí ho \C zachránit.\quad\X
}

\sloka{
\GSiroky Otři si \D slzy dobře \Emi víš, že pláč ten nic už \C nespraví\\
\GSiroky musíš teď \D jít, ne takhle \C stát.\quad\X\\
\GSiroky Nezou\D fej, dej si \Emi radit pohádkou tu už \C víc nikdo nezmění\\
\GSiroky osud ti \D dá co musí \C dát.\quad\X
}

\refren{}\vspace{-1em}

\sloka{
\GSiroky Bude to \D boj srdce z \Emi ledu, její cit dobře \C víš kdo v něm zvítězí.\\
\GSiroky Vzpomene \D Káj a jeho srdce \C roztaje\quad\X\\
\GSiroky a stejně \D ty, jdi teď \Emi za ní je v tvých silách \C tu druhou porazit\\
\GSiroky tak jako \D Gerda, když šla za \C Kájem.\quad\X
}
