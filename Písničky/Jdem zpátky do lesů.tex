\zvyraznit
\hlavicka[Žalman]{Jdem zpátky do lesů}

\sloka{
\AmiSedm Sedím na kolejích, \DSedm které nikam neve\G dou,\quad\X\\
\AmiSedm koukám na kopretinu, jak \DSedm miluje se s lebe\G dou.\quad\X\\
\AmiSedm Mraky vzali slunce, \DSedm zase pod svou ochra\G nu,\quad\Emi\\
\AmiSedm jen ty nejdeš holka zlatá, \DSedm kdypak já tě dosta\G nu.\quad\DSedm
}

\refren{
\G Z ráje,\quad\X\quad my vyhnaní \Emi z ráje,\\
\X\quad kde není už \AmiSedm místa,\quad\C\quad prej něco se \G chystá.\quad\DSedm\\
\G Z ráje,\quad\X\quad nablýskaných \Emi plesů,\\
\X\quad jdem zpátky do \AmiSedm lesů,\quad\C\quad za nějaký \G čas.\quad\DSedm
}

\sloka{
Vlak nám včera ujel, za stanice do nebe,\\
málo jsi se snažil, málo jsi šel do sebe.\\
Šel jsi vlastní cestou a to se dneska nenosí\\
i pes, kterej chce přízeň, napřed svýho pána poprosí.
}

\refren{}

\sloka{
Už tě vidím zdálky, jak máváš na mě korunou,\\
jestli nám to bude stačit, zatleskáme na druhou.\\
Zabalíme všechny, co si dávaj rande za branou,\\
v ráji není místa, možná v pekle se nás zastanou.
}

\refren{}
