\hlavicka{Chodím Po Broadwayi}{}


\sloka{
Chodím \D po Broadwayi hladov sem a tam,\\
\D chodím po Broadwayi hladov sem a \A tam,\\
chodím \D po Broadwayi, \G po Broadwayi,\\
\D po Broadwayi \A hladov sem a \D tam.\\
}

\refren{
\D Singi jou jou jupí jupí jou,\\
\D singi jou jou jupí jupí \A jou,\\
singi \D jou jou jupí, \G jou jou jupí,\\
\D singi jou jou \A jupí jupí \D jou.
}

\sloka{
Práci nedostanu, černou kůži mám,\\
práci nedostanu, černou kůži mám,\\
práci nedostanu, nedostanu,\\
protože já černou kůži mám.
}

\refren{Singi jou jou jupí\ldots}

\sloka{

moje černé děti mají stále hlad,\\
moje černé děti, černé děti,\\
černé děti mají stále hlad.
}

\refren{Singi jou jou jupí\ldots}
