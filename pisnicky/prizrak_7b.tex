\hlavicka{Přízrak 7.B}{Karamel}

\sloka{
Její \G tvář všem školním fotkám vévodí,\\
dobře \Emi zná, co se má a co se nehodí,\\
jenom \C pár slov s námi občas prohodí,\\
pyšná jak\G páv. \D 
}

\sloka{
Každý kluk ztrácí řeč, tají dech ani nedutá,\\
jde dál školní chodbou děsně  nadutá,\\
školní řád překročit, obejít,\\
jen ať všimne si nás.
}

\refren{
\repetice{
Dál sama si \Emi kráčí po cestách \G svých,\\
je jiná, je \Emi zvláštní než ty z jiných \G tříd.\D 
}}

\sloka{
Školní sbor vede spor o jejím prospěchu,\\
zda se smí v hlavách kluků dělat neplechu,\\
sklopí zrak, tak že školník se dá do breku,\\
a ten toho ví.
}

\sloka{
Drží krok s módou, to jí holky závidí,\\
jak je štíhlá a snědá se jim nelíbí,\\
ale nám, klukům spát to nedá, co kdyby,\\
všimla si nás.
}
