\hlavicka{Neříkej}{Reflexy}

\sloka{
\D Déšť ti slzy \A utírá, a \G v srdci bolest \D máš\\
\D láska se ti \A vyhýbá, když \G marně ji hle\D dáš.
}

\refren{
No tak \G pojď, já se \D trápím, no tak \G pojď má malá \D paní\\
no tak \G pojď já tě \D prosím pojď se \A mnou.\\
Neří\D kej, že to láska není neří\Hmi kej, že v tobě není\\
neří\G kej, že láska bolí, ne, \D ne to neří\A kej\\
Líbej \D mě, už nechci lhát, líbej \Hmi mě ústy, co pálí\\
líbej \G mě já mám tě rád, líbej \A mě a ty mě líbáš a ty mě líbáš, mám tě \D rád.
}

\sloka{
Déšť ti slzy nestírá a v srdci lásku máš\\
těžko se ti usíná, když víc už nehledáš.
}

\refren{No tak pojď \ldots}
