\hlavicka{Barák na vodstřel}{Aleš Brichta}

\sloka{  
  \Emi V tom šeru \D dvorů trpěly \Ami místy zbytky \C trávy\\
  \Emi z pavlačí \D dolů visely \Ami stonky zvadlejch \C květin\\
  \Emi Jen vokna zapráše\D ný s čarami \Ami teď už prošlý \C zprávy\\
  \Emi dvě srdce nakresle\D ný písmena \Ami co měl dávno \C smazat \Emi déšť\\
 }
\refren{
  Tenhle \C barák je na vod\G střel je dopis \D promlčený \Ami lásky\\
  Tenhle \C barák je na vod\G střel kam večer \D chodilo se na zeď\Ami psát\\
  Tenhle \C barák je na vod\G střel co odpovídat \D měl nám na \Ami otázky\\
  Tenhle \C barák je na vod\G střel už nebudeš \D se jeho \Ami stínů \Emi ptát\\
  }
\sloka{
  Dav čumilů civí těší se až ta barabizna zmizí\\
  kdekdo se diví že tu mohla takhle dlouho stát\\
  Vzpomínky mávaj a Ty procházíš svou malou krizí\\
  Času se nedostává - na pohled zbejvá už jen vteřin pár\\
  }
\refren{
  Tenhle barák je na vodstřel a na Tvý tváři pár let starý vrásky\\
  Tenhle barák je na vodstřel kde ze zdí čiší vyklizenej chlad\\
  Tenhle barák je na vodstřel co půvab má už jenom pro obrázky\\
  Tenhle barák je na vodstřel proč najednou už se Ti vůbec nechce smát ?\\
}
