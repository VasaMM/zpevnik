\hlavicka{Já si kopu vlastní hrob}{Kabát}

\sloka{
Já si kopu vlastní hrob,\\
zbejvá mi už jen deset stop,\\
krumpáčem, lopatou,\\
večer dám si rande s paní zubatou.
}

\sloka{
\Fis Neměj \A žádnou \Ha obavu\D ,\\
\Fis tebe a \A tu h\Ha línu hodim \D za hlavu,\\
\Fis najdi si \A nový\Ha ho,\\
\A mladýho \E zdravýho.
}

\refren{
Já hodně pil a hodně klel,\\
rychle žil a vůbec nevěděl,\\
že si kopu vlastní hrob,\\
už mi zbejvá jenom devě stop.
}

\sloka{
A brouci plný naděje,\\
dělaj jako že se nic neděje,\\
a už maj chuť si potykat,\\
a tykadlama jemně se mě dotýkat.
}

\sloka{
Největší z nich hovnivál,\\
sed na kouli-koulel oči povídal,\\
neměls to přehánět,\\
teď si budem v tobě pěkně dovádět.
}
