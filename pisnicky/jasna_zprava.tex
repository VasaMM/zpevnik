\hlavicka{Jasná zpráva}{Olympic}

\sloka{
\G Skončili jsme jasná zpráva, \\
\Emi proč o tebe \C zakopávam \D dál, \\
\Ami projít bytem \C já abych se \G bál. \\
\G Dík tobě se vidím zvenčí, \\
\Emi připadám si \C starší menší \D sám, \\
\Ami kam se kouknu, \C kousek tebe \G mám\\
}

\refren{
\Emi Pěnu s vůni \G jablečnou, \Emi vyvanulý \Hmi sprej, \\
\Emi telefon, cos \G ustřihla mu \D šňůru, \\
\Emi knížku krásně \G zbytečnou, \Emi co má lživý \Hmi děj, \\
\Emi píše se v ní, \G jak se lítá \D vzhůru,\\ 
lítá \Ami vzhůru, ve dvou \D vzhůru
}

\sloka{
\G Odešla's mi před úsvitem, \\
\Emi mám snad bloudit \C vlastním bytem \D sám,\\ 
\Ami kam se kouknu, \C kousek tebe \G mám
}

\refren{Pěnu s vůní jablečnou \ldots}
 
\sloka{
\G Skončili jsme jasná zpráva, \\
\Emi není komu z \C okna mávat \D víc,\\ 
\Ami jasná zpráva, \C rub, co nemá \G líc
}
