\hlavicka{Piju já piju rád}{Kabát}

\sloka{        
  \Hmi Piju já piju rád\\
  destilát, ten \G umim pálit\\
  \Hmi ochutnej, škoda slov\\
  tohle nejde \G nepochválit\\
  \D srdce jásá mozek usíná\\
  kdo \A pije tak i po smrti se usmívá
}
 
\sloka{ 
  Překapat, filtrovat\\
  ejhle je tu medicína\\
  potom nám nehrozí\\
  v létě úpal, v zimě rýma\\
  četník za mnou přišel, já ho nepozval\\
  a už mě vítá brána, za ní kriminál
}
\refren{
  Co ch\Hmi těj ty stráže\\
  proč \G železa mi dá\D vaj, \A já jenom pil\Hmi\\
  \Hmi jak zákon káže\\
  \G já ten jejich \D zákon \A zle pochopil\Hmi\\
   co \Hmi chtěj ty stráže\\
  proč \G železa mi dávaj\D, A já jenom \Hmi pil\\
  \Hmi jak zákon káže\\
  \G já ten jejich \D zákon \A zle pochopil\Hmi
}
\sloka{
  Vážnej pán, s parukou\\
  kladivem do stolu mlátil\\
  prosím klid, povídá\\
  prej abych všechno státu vrátil\\
  tak mu říkám, já nic neurad\\
  pálil jsem jen proto, že to piju rád
}
