\hlavicka{Slibuju že nebudu pít}{Rybičky 48}

\begin{multicols}{2}
\sloka{
Je deset hodin \A  ráno,
\\
zalepený \CisMi voči, nemůžu vstát.
\\
Kafe a \D cigáro si dám,
\\
včera jsem \Dmi byl zas jako Dán.\\


A ty se díváš, 
\\
ty se na mě díváš pohledem,
\\
kterej každej mužskej zná,
\\
a jen žena ho dovede.
}

\refren{
\A  Chceš ode mě slyšet tu větu vět,
\\
kterou i \CisMi Shakespeare znal nazpaměť.
\\
Kterou i \D Adam říkal Evě,
\\
když večer \F předtím zašel na dvě.
\\

Chceš ode mě slyšet tu větu vět,
\\
já nechci kázání, ani jít na zpověď.
\\
Uchýlim se k malé lži,
\\
snad karma promine, vždyť jsme jen muži.
\\

\repetice{
\A Slibuju, že nebudu \E pít,
\\
\FisMi včera to bylo \D naposled.
\\
\A Slibuju, že nebudu \E pít,
\\
to přísahám\D .}}

\sloka{
Kde jsem včera byl,
\\
jak jsem se asi měl.
\\
Snad kamarádi ví,
\\
když já to zapomněl.
\\

V krku mě bolí,
\\
takže se asi zpívalo.
\\
A když dneska smutno je,
\\
tak včera asi bylo veselo.
}

\refren{Chceš ode mě slyšet tu větu vět \ldots}

\refren{
Slibuju, že nebudu pít,
\\
včera to bylo naposled.
\\
Slibuju, že nebudu pít,
\\
to přísahám.
\\

\A Už jenom na svatbách,
\\
a nebo \E promocích.
\\
\FisMi Přes den ne,
\\
jenom po \D nocích.
\\

\A Na oslavách,
\\
nebo \E večírcích.
\\
\FisMi Přes den ne,
\\
jenom \D po nocích.

\repetice{\A Slibuju, že nebudu \E pít, to přísá\D hám.}}
\end{multicols}
